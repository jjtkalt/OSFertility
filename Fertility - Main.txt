string getAllSaveDataForDB()
{
    string data = "";
    if(gender == -1) { return [-1]; }
    data += "Name="+ (string)llKey2Name(Owner);
    data += "|TOD="+ (string)now;
    data += "|Gender="+ (string)gender;
    if(gender == 1) {
        // Female
        data += "|CycleStage="+ (string)cycleStage;
        string theseDonors = "";
        string theseExpires = "";
        if ( llGetListLength(spermDonors) > 1) {
            theseDonors = llDumpList2String(spermDonors,[","]);
            theseExpires = llDumpList2String(spermExpires,[","]);
        }
        else if ( llGetListLength(spermDonors) == 1) {
            theseDonors = llList2String(spermDonors,0);
            theseExpires =llList2String(spermExpires,0);
        }
        else if ( llGetListLength(spermDonors) < 1) {
            theseDonors = "None";
            theseExpires = "None";
        }
        data += "|Donors="+ theseDonors;
        data += "|Expires="+ theseExpires;
        
        if(cycleStage == -1) {
            //Pregnant
            if(babyDaddyKey != "" && babyDaddyKey != "None")
            {
                data += "|BabyDaddy="+ (string)babyDaddyKey;
                data += "|TermDay="+ (string)pregnancyDay;
            }
        }
        else {
            data += "|BabyDaddy=None";
            data += "|TermDay=None";
        }
    }
    
    if(debugSpam == TRUE) { llOwnerSay("Changes == "+(string) data); }
    return data;
}

loadFromDB(list Values)
{
    integer i;
    string debugingDB = "";
    for(i=0; i<llGetListLength(Values); i++) {
        list Value = llParseString2List(llList2String(Values,i), ["="], [""]);
        if(debugSpam == TRUE) { llOwnerSay("DB: "+llDumpList2String(Value, ["="])); }
        string Key = llToLower(llList2String(Value, 0));
        string KeyValue = llToLower(llList2String(Value, 1));
        if(Key == "name") {
            Name = KeyValue;
            debugingDB += Name + " | " + KeyValue;
        }
        else if(Key == "gender") {
            gender = (integer)KeyValue;
            debugingDB += (string)gender + " | " + KeyValue;
        }
        else if(Key == "tod"){
            now = (integer)KeyValue;
            debugingDB += (string)now + " | "+ KeyValue;
        }
        else if(Key == "donors") {
            if(KeyValue == "" || KeyValue == "None" ) {
                spermDonors = [];
                debugingDB += llDumpList2String(spermDonors, ",") + " | "+ "''";
                spermExpires = [];
                debugingDB += llDumpList2String(spermExpires, ",") + " | "+ "None";
            }
            else {
                list theseSperm = llParseString2List(KeyValue,[","],[""]);
                if(llGetListLength(theseSperm) <= 1) { spermDonors =  llList2String(Value, 1); }
                else {
                    integer i = 0;
                    while(i < llGetListLength(theseSperm) )
                    {
                        if(llList2String(theseSperm, i) != "" && llList2String(theseSperm, i) != "None" ) spermDonors += llList2List(theseSperm, i, i);
                        i++;
                    }
                }
                debugingDB += (string)llGetListLength(spermDonors) + " " + llDumpList2String(spermDonors, ",") + " | " + KeyValue;
            }
        }
        else if(Key == "expires") {
            if(KeyValue == "" || KeyValue == "None" ) {
                spermExpires = [];
                debugingDB += llDumpList2String(spermExpires, ",") + " | " + "''";
            }
            else {
                list theseExpires = llParseString2List(KeyValue,[","],[""]);
                if(llGetListLength(theseExpires) <= 1) { spermExpires =  KeyValue; }
                else {
                    integer i = 0;
                    while(i < llGetListLength(theseExpires) ) {
                        if(llList2String(theseExpires, i) != "" && llList2String(theseExpires, i) != "None" ) spermExpires += llList2List(theseExpires, i, i);
                        i++;
                    }
                }
                debugingDB += (string)llGetListLength(spermExpires) + " " + llDumpList2String(spermExpires,",") + " | " + KeyValue;
            }
        }
        else if(Key == "babydaddy") {
            babyDaddyKey = KeyValue;
            debugingDB += (string)babyDaddyKey + " | " + KeyValue;
        }
        else if(Key == "termday") {
            pregnancyDay = (integer)KeyValue;
            debugingDB += (string)pregnancyDay + " | " + KeyValue;
        }
        else if(Key == "cyclestage") {
            cycleStage = (integer)KeyValue;
            debugingDB += (string) cycleStage + " | " + KeyValue;
        }
        debugingDB += "\n";
    }
    if(debugSpam == TRUE) { llOwnerSay("Changes == "+(string) debugingDB); }
}

integer getLinkFromList(string name)
{
    integer link = llListFindList(linkedButtons, [name]);
    if( link == -1) { link = 999 };
    return link+2;
}

integer checkMenuNeeded()
{
    getActionsMenu();
    if(llListFindList(usersTargetingMe,currentTarget) != -1 || buttons != []) { return TRUE; }
    return FALSE;
}

list getActionsMenu()
{
    buttons = [];
    if(currentTarget != NULL_KEY)
    {
        if(llListFindList(usersTargetingMe, currentTarget) == -1)
        {
            buttons += ["Force"];
        }
    }
    if(llListFindList(usersTargetingMe, currentTarget) != -1)
    {
        if(gender == 0)
        {
            buttons += ["Check"];
            buttons += ["Cum Inside"];
        }
        if(gender == 1) buttons += ["Orgasm"];
    }
    return buttons;
}
updateLinkAlphas()
{
    llSetLinkAlpha(getLinkFromList(llGetObjectName()),1,ALL_SIDES);
    llSetLinkAlpha(getLinkFromList("fertility Hud"),1,ALL_SIDES);
    llSetLinkAlpha(getLinkFromList("Menu"),checkMenuNeeded(),ALL_SIDES);
    if(Settings > 0) { llSetLinkAlpha(getLinkFromList("Settings"),1,ALL_SIDES); }
    else if(Settings <= 0) { llSetLinkAlpha(getLinkFromList("Settings"),0,ALL_SIDES); }
    if(onlineMode != -1) {
        llSetLinkAlpha(getLinkFromList("Online"),onlineMode,ALL_SIDES);
        llSetLinkAlpha(getLinkFromList("Offline"),!onlineMode,ALL_SIDES);
    }
    if(gender == 0) {
        // Show Male things
        if(llGetListLength(otherUserKey) == 0) { llSetLinkAlpha(getLinkFromList("Girl"),0,ALL_SIDES); }
        else { llSetLinkAlpha(getLinkFromList("Girl"),1,ALL_SIDES); }
        if(usersTargetingMe == []) { llSetLinkAlpha(getLinkFromList("girl Hearts"),0,ALL_SIDES); }
        else { llSetLinkAlpha(getLinkFromList("girl Hearts"),1,ALL_SIDES); }
    }
    else if(gender == 1) {
        // Show Female things
        if(showCycle == 0) {
            integer i;
            for(i=0; i<6; i++) {
                if(i==4 || i== 5) { llSetLinkAlpha(getLinkFromList("Cycle"),1,i); }
            }
        }
        else { llSetLinkAlpha(getLinkFromList("Cycle"),1,ALL_SIDES); }
        //set cycle link rotation
        if(llGetListLength(otherUserKey) == 0) { llSetLinkAlpha(getLinkFromList("Boy"),0,ALL_SIDES); }
        else { llSetLinkAlpha(getLinkFromList("Boy"),1,ALL_SIDES); }
        if(usersTargetingMe == []) { llSetLinkAlpha(getLinkFromList("boy Hearts"),0,ALL_SIDES); }
        else { llSetLinkAlpha(getLinkFromList("boy Hearts"),1,ALL_SIDES); }
    }

    if(currentTarget == NULL_KEY) { llSetText("",<1,1,1>,1); }
    else {
        integer i;
        if(llListFindList(usersTargetingMe,currentTarget) != -1 ) { llSetText("❤❤ "+llGetDisplayName(currentTarget)+" ❤❤", <1,1,1>,1); }
        else { llSetText(">> "+llGetDisplayName(currentTarget), <1,1,1>,1); }
    }
}

    {
        if( id == Owner ) {
            if(initialize == 0) {
                if(m == "Online") {
                    initialize = 1;
                    onlineMode = 1; //online mode
                    username = Owner;
                    password = "";
                    llTextBox(Owner, "Type in your password:", menuChannel);
                    return;
                }
                else if(m == "Offline") {
                    initialize = 2;
                    onlineMode = 0; //offline mode
                    llOwnerSay("Offline mode chosen\nWARNING: Your HUD progress may get reset if the script is reset such as during hyper-griding.");

                        keyConfigQueryhandle = llGetNotecardLine(NOTECARD, intLine1);
                        keyConfigUUID = llGetInventoryKey(NOTECARD);
                    }
                    else { llDialog(Owner,"Gender?", ["Male", "Female"], menuChannel); }
                    return;
                } 
            }
            else if(initialize == 1) {
                if(onlineMode == 1 && password == "") {
                    password = llEscapeURL(m);
                    http_request_id = llHTTPRequest(url, [HTTP_METHOD, "POST", HTTP_MIMETYPE, "text/plain"], "UN=" + username + "|PW=" + password + "|CM=Load:Name|Gender|TOD|Donors|Expires|BabyDaddy|TermDay|CycleStage");
                }
            }
            else if(initialize == 2) {
                if(m == "Male") {
                    gender = 0;
                    llOwnerSay("You chose male.")
                }
                else if(m == "Female") {
                    gender = 1;
                    llOwnerSay("You chose female.")
                }
                if(gender != -1)state running;
            }
        }
    }

    http_response(key request_id, integer status, list metadata, string body)
    {
        if (request_id != http_request_id) { return; }
        if(debugSpam == TRUE) { llOwnerSay("ID: " + (string)request_id + " Status: " + (string)status + " MetaData: " + llDumpList2String(metadata, "|") + " Body: " + body); }
        if (status == 499) {
            llOwnerSay("Connection was refused. Switching to Offline Mode.");
            onlineMode = FALSE;
            //Get Settings
            initialize=2;
            if (llGetInventoryType(NOTECARD) != INVENTORY_NONE) {
                keyConfigQueryhandle = llGetNotecardLine(NOTECARD, intLine1);
                keyConfigUUID = llGetInventoryKey(NOTECARD);
            }
            else { llDialog(Owner,"Gender?", ["Male", "Female"], menuChannel); }
            return;
        }
        
        
        if(body == "UserNotExists") { http_request_id = llHTTPRequest(url, [HTTP_METHOD, "POST", HTTP_MIMETYPE, "text/plain"], "UN=" + username + "|PW=" + password + "|CM=new:"); }
        else if(body == "ErrorUserPwd") {
            //wrong password try again
            password = "";
            llTextBox(Owner, "Wrong password, Type in your password:", menuChannel);
            return;
        }
        list bits = llParseString2List(body, ["|"],[""]);
        if(llList2String(bits,0)  == "CM=New") {
            if(llList2String(bits,1) == "okNewUser") {
                initialize=2;
                isNew = TRUE;
                 //Get Settings
                if (llGetInventoryType(NOTECARD) != INVENTORY_NONE) {
                    keyConfigQueryhandle = llGetNotecardLine(NOTECARD, intLine1);
                    keyConfigUUID = llGetInventoryKey(NOTECARD);
                }
                else { llDialog(Owner,"Gender?", ["Male", "Female"], menuChannel); }
                return;
            }
        }
        else if(llList2String(bits,0)  == "CM=Load") {
            if(llList2String(bits,1) == "Name=None") {
                initialize=2;
                isNew = TRUE;
            }
            else {
                isNew = FALSE;
                loadFromDB(bits);
                llOwnerSay("Loaded your account from DB.");
            }
            //gender = llList2Integer(bits,1);
            keyConfigQueryhandle = llGetNotecardLine(NOTECARD, intLine1);
            keyConfigUUID = llGetInventoryKey(NOTECARD);
        }
    }

    dataserver(key keyQueryId, string strData)
    {
        if (keyQueryId == keyConfigQueryhandle) {
            if (strData == EOF) {
                llOwnerSay("Loaded Settings from notecard.");
                if(gender == -1) {
                    //if you reach eof and still dont know gender then ask for it.
                    llDialog(Owner,"Gender?", ["Male", "Female"], menuChannel);
                    return;
                }
                else { state running; }
            }

            keyConfigQueryhandle = llGetNotecardLine(NOTECARD, ++intLine1);

            if (llGetSubString (strData, 0, 0) != "#") {
                list data = llParseString2List(strData,["="],[]); //split around the =
                if(debugSpam == TRUE) { llOwnerSay("NC: " + llDumpList2String(data, ["="])); }
                string field = llList2String(data,0);
                if(field == "gender" && gender == -1) { gender = llList2Integer(data,1); }
                else if(field == "showCycle") { showCycle = llList2Integer(data,1); }
                else if(field == "fertilityChance") { fertilityChance = llList2Integer(data,1); }
                else if(field == "maxDonors") { maxDonors = llList2Integer(data,1); }
                else if(field == "dayLength") { dayLength = llList2Integer(data,1); }
                else if(field == "pregnancyTermLengthInDay") { pregnancyTermLengthInDay = llList2Integer(data,1); }
                else if(field == "onlyPartner") {
                    if(llList2Integer(data,1) != -1) { partnerKey = llList2Key(data,1); }
                    else { partnerKey = NULL_KEY; }
                }
                else if(field == "allowRLV") {
                    if(llList2Integer(data,1) == 1) { canRLV = TRUE; }
                    else { canRLV = FALSE; }
                }
                else if(field == "allowForce") {
                    // Enables the force features. >off by default<  0 off 1 on
                    if(llList2Integer(data,1) == 1) { canForce = TRUE; }
                    else { canForce = FALSE; }
                }
                else if(field == "cycleTimeOvulate") { nextCycleEventTime = llListReplaceList(nextCycleEventTime,llList2Integer(data,1),0,0); }
                else if(field == "cycleTimeFertile") { nextCycleEventTime = llListReplaceList(nextCycleEventTime,llList2Integer(data,1),1,1); }
                else if(field == "cycleTimePeriod") { nextCycleEventTime = llListReplaceList(nextCycleEventTime,llList2Integer(data,1),2,2); }
                else if(field == "cycleTimeRecovery") { nextCycleEventTime = llListReplaceList(nextCycleEventTime,llList2Integer(data,1),3,3); }
            }
        }
    }

                    cycleDayTimer=0;
                }
            }
            if(offlineRPDays != 0)
            {
                if(offlineRPDays > 1)llOwnerSay(offlineRPDays+" RP days have passed.");
                offlineRPDays=0;
                saveTick = 100;
            }
            //female end
        }
    }
